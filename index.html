        function calculateKPI() {

            const connectRate = parseFloat(document.getElementById("connectRate").value);

            const conversationRate = parseFloat(document.getElementById("conversationRate").value);

            const meetingRate = parseFloat(document.getElementById("meetingRate").value);

            const meetingGoal = parseInt(document.getElementById("meetingGoal").value);

            const businessDays = parseInt(document.getElementById("businessDays").value);

            const callSteps = parseInt(document.getElementById("callSteps").value);

            const sequenceDuration = parseInt(document.getElementById("sequenceDuration").value);

            const callStepFrequency = parseInt(document.getElementById("callStepFrequency").value);

            const currentProspects = parseInt(document.getElementById("currentProspects").value);

            const meetingShowRatio = parseFloat(document.getElementById("meetingShowRatio").value);


            if (

                isNaN(connectRate) ||

                isNaN(conversationRate) ||

                isNaN(meetingRate) ||

                isNaN(meetingGoal) ||

                isNaN(businessDays) ||

                isNaN(callSteps) ||

                isNaN(sequenceDuration) ||

                isNaN(callStepFrequency) ||

                isNaN(currentProspects) ||

                isNaN(meetingShowRatio) ||

                connectRate < 0 || connectRate > 100 ||

                conversationRate < 0 || conversationRate > 100 ||

                meetingRate < 0 || meetingRate > 100 ||

                callSteps < 1 ||

                sequenceDuration < 1 ||

                (callStepFrequency !== 1 && callStepFrequency !== 2 && callStepFrequency !== 3) ||

                currentProspects < 0 ||

                meetingShowRatio < 0 || meetingShowRatio > 100

            ) {

                alert("Please enter valid values.");

                return;

            }


            // Suggestions for improving input values

            let suggestions = [];


            if (connectRate < 10) {

                suggestions.push("Consider improving your outreach strategies to increase connect rates.");

            }


            if (conversationRate < 20) {

                suggestions.push("Focus on improving your messaging and engagement techniques to increase conversation rates.");

            }


            if (meetingRate < 5) {

                suggestions.push("Enhance your meeting scheduling and follow-up processes to boost meeting conversion rates.");

            }


            // Display suggestions

            const suggestionsDiv = document.getElementById("suggestions");

            suggestionsDiv.innerHTML = ""; // Clear previous suggestions


            if (suggestions.length > 0) {

                suggestionsDiv.innerHTML = "<p>Suggestions to Improve:</p><ul>";

                suggestions.forEach(suggestion => {

                    suggestionsDiv.innerHTML += `<li>${suggestion}</li>`;

                });

                suggestionsDiv.innerHTML += "</ul>";

            } else {

                suggestionsDiv.innerHTML = "<p>Your input values meet industry standards.</p>";

            }


            // Calculations and results

            const adjustedMeetingRate = meetingRate * (meetingShowRatio / 100);

            const requiredDialsPerDay = (meetingGoal / (connectRate / 100) / (conversationRate / 100) / (adjustedMeetingRate / 100) / businessDays).toFixed(1);

            const requiredMeetingsPerDay = (requiredDialsPerDay * (connectRate / 100) * (conversationRate / 100) * (adjustedMeetingRate / 100)).toFixed(1);

            const callsPerSequence = callSteps * (sequenceDuration / callStepFrequency);

            const requiredSequencesPerDay = (requiredDialsPerDay / callsPerSequence).toFixed(1);


            let prospectsToSequenceDaily = (requiredSequencesPerDay - currentProspects).toFixed(1);

            prospectsToSequenceDaily = Math.max(prospectsToSequenceDaily, 0);

            const minProspectsToMaintainSequences = (currentProspects + (requiredSequencesPerDay * businessDays)).toFixed(1);


            document.getElementById("requiredDialsDaily").textContent = `You need to make approximately ${requiredDialsPerDay} dials per business day to achieve your Monthly Meeting KPI.`;

            document.getElementById("requiredMeetingsDaily").textContent = `This will result in approximately ${requiredMeetingsPerDay} meetings set per business day.`;


            if (prospectsToSequenceDaily === "0.0") {

                document.getElementById("requiredSequencesDaily").textContent = `To maintain leads, you should have at least ${minProspectsToMaintainSequences} prospects in sequence each business day.`;

            } else {

                document.getElementById("requiredSequencesDaily").textContent = `To maintain leads, you should sequence approximately ${prospectsToSequenceDaily} prospects per business day.`;

            }


            const meetingMultiplier1 = (meetingGoal * 1.5).toFixed(1);

            const meetingMultiplier2 = (meetingGoal * 2).toFixed(1);

            const meetingMultiplier3 = (meetingGoal * 3).toFixed(1);


            document.getElementById("meetingMultiplier").innerHTML = `

                To achieve:<br>

                - 1.5x your meeting quota, aim for approximately ${meetingMultiplier1} meetings per month.<br>

                - 2x your meeting quota, aim for approximately ${meetingMultiplier2} meetings per month.<br>

                - 3x your meeting quota, aim for approximately ${meetingMultiplier3} meetings per month.

            `;


            const dialsForMultiplier1 = (meetingMultiplier1 / (connectRate / 100) / (conversationRate / 100) / (adjustedMeetingRate / 100) / businessDays).toFixed(1);

            const dialsForMultiplier2 = (meetingMultiplier2 / (connectRate / 100) / (conversationRate / 100) / (adjustedMeetingRate / 100) / businessDays).toFixed(1);

            const dialsForMultiplier3 = (meetingMultiplier3 / (connectRate / 100) / (conversationRate / 100) / (adjustedMeetingRate / 100) / businessDays).toFixed(1);


            document.getElementById("dialsForMultiplier").innerHTML = `

                To achieve:<br>

                - 1.5x your meeting quota, you need to make approximately ${dialsForMultiplier1} dials per business day.<br>

                - 2x your meeting quota, you need to make approximately ${dialsForMultiplier2} dials per business day.<br>

                - 3x your meeting quota, you need to make approximately ${dialsForMultiplier3} dials per business day.

            `;

        }


        function clearFields() {

            const inputFields = document.querySelectorAll('input[type="number"]');

            inputFields.forEach(field => field.value = "");


            const suggestionsDiv = document.getElementById("suggestions");

            suggestionsDiv.innerHTML = "";

        }
